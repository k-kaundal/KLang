# AI-Powered Web Scraper with Analysis
import ai
import http

function scrape_and_analyze(url) {
    # Fetch web content
    let response = http.get(url)
    let content = response.text
    
    # Use AI to extract key information
    let summary = ai.llm(`
        Summarize this webpage in 3 bullet points:
        ${content}
    `)
    
    # Generate embeddings for semantic search
    let embedding = ai.embed(summary)
    
    # Use RAG for context-aware responses
    let rag = ai.rag({
        documents: [content],
        query: "What are the main topics?"
    })
    
    return {
        summary: summary,
        embedding: embedding,
        topics: rag
    }
}

# Create an AI agent
let agent = ai.agent({
    name: "WebAnalyzer",
    goal: "Analyze web content",
    tools: ["http", "summarizer"],
    memory: true
})

# Let the agent work autonomously
agent.run("Analyze https://example.com and provide insights")

# Async/Await for Concurrent Operations
import http
import time

async function fetch_data(url) {
    println("Fetching:", url)
    let response = await http.get(url)
    return response.json()
}

async function fetch_all() {
    # Concurrent requests
    let promises = [
        fetch_data("https://api1.example.com/data"),
        fetch_data("https://api2.example.com/data"),
        fetch_data("https://api3.example.com/data")
    ]
    
    # Wait for all to complete
    let results = await Promise.all(promises)
    return results
}

# Async generator
async function* generate_data() {
    for i in range(5) {
        await time.sleep(1)
        yield i * 2
    }
}

# Main async function
async function main() {
    let data = await fetch_all()
    println("All data fetched:", data)
    
    # Use async generator
    async for value in generate_data() {
        println("Generated:", value)
    }
}

# Run async code
main()

# Object-Oriented Programming in KLang
class Animal {
    constructor(name, age) {
        this.name = name
        this.age = age
    }
    
    speak() {
        println(this.name + " makes a sound")
    }
    
    get info() {
        return `${this.name} is ${this.age} years old`
    }
}

class Dog extends Animal {
    constructor(name, age, breed) {
        super(name, age)
        this.breed = breed
    }
    
    speak() {
        println(this.name + " barks!")
    }
    
    fetch() {
        println(this.name + " fetches the ball")
    }
}

# Using classes
let dog = new Dog("Max", 3, "Golden Retriever")
dog.speak()      # Max barks!
dog.fetch()      # Max fetches the ball
println(dog.info) # Max is 3 years old

# Class with AI capabilities
class SmartAssistant extends Animal {
    constructor(name) {
        super(name, 0)
        this.ai = ai.llm_client()
    }
    
    async think(question) {
        let response = await this.ai.ask(question)
        return response
    }
}

let assistant = new SmartAssistant("Jarvis")
let answer = await assistant.think("What's the weather?")
println(answer)

# HTTP Client with Full Features
import http
import json

# Simple GET request
let response = http.get("https://api.example.com/data")
println(response.status)  # 200
println(response.json())

# POST request with headers
let post_response = http.post(
    "https://api.example.com/users",
    {

            "Authorization": "Bearer token123"
        },
        body: json.dumps({
            name: "John Doe",

    }
)

# Error handling
try {
    let data = http.get("https://invalid-url.com")
} catch (error) {
    println("Request failed:", error.message)
}

# Custom HTTP client
class APIClient {
    constructor(base_url, api_key) {
        this.base_url = base_url
        this.api_key = api_key
    }
    
    request(method, endpoint, data = null) {
        let url = this.base_url + endpoint
        let options = {
            method: method,
            headers: {
                "Authorization": "Bearer " + this.api_key
            }
        }
        
        if (data) {
            options.body = json.dumps(data)
            options.headers["Content-Type"] = "application/json"
        }
        
        return http.request(url, options)
    }
    
    get(endpoint) {
        return this.request("GET", endpoint)
    }
    
    post(endpoint, data) {
        return this.request("POST", endpoint, data)
    }
}

# Using custom client
let client = new APIClient(
    "https://api.example.com",
    "your-api-key"
)

let users = client.get("/users")
let new_user = client.post("/users", {
    name: "Jane Doe"
})
