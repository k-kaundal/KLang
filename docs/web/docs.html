KLang Documentation

Welcome to the complete documentation for KLang, the AI-native programming language. This guide will help you learn everything from basic syntax to advanced AI integration.

Installation

Getting started with KLang is easy. Follow these steps to install on your system:

Linux/macOS

# Clone the repository
git clone https://github.com/k-kaundal/KLang.git
cd KLang

# Build and install
make
sudo make install

# Verify installation
klang --version

Windows

# Using WSL (recommended)
git clone https://github.com/k-kaundal/KLang.git
cd KLang
make
sudo make install

# Or download pre-built binaries
# Visit: https://github.com/k-kaundal/KLang/releases
Note: KLang requires GCC or Clang compiler and Make build tool. Make sure these are installed on your system before building.

Hello World

Let's write your first KLang program:

# hello.kl
println("Hello, World!")

Functions

# Regular function
function add(a, b) {
    return a + b
}

# Arrow function
let multiply = (a, b) => a * b

# Function with default parameters
function greet(name = "World") {
    return "Hello, " + name
}

# Rest parameters
function sum(...numbers) {
    let total = 0
    for num in numbers {
        total += num
    }
    return total
}

# Example usage
println(add(5, 3))        # 8
println(multiply(4, 7))   # 28
println(greet())          # Hello, World
println(sum(1, 2, 3, 4))  # 10

Classes and OOP

# Class definition
class Person {
    constructor(name, age) {
        this.name = name
        this.age = age
    }
    
    greet() {
        println("Hello, my name is " + this.name)
    }
    
    # Getter
    get info() {
        return this.name + " is " + str(this.age) + " years old"
    }
    
    # Static method
    static create(name, age) {
        return new Person(name, age)
    }
}

# Inheritance
class Student extends Person {
    constructor(name, age, grade) {
        super(name, age)
        this.grade = grade
    }
    
    study() {
        println(this.name + " is studying")
    }
}

# Usage
let person = new Person("Alice", 25)
person.greet()
println(person.info)

let student = new Student("Bob", 20, "A")
student.greet()
student.study()

Async/Await

# Async function
async function fetchData(url) {
    let response = await http.get(url)
    return response.json()
}

# Error handling with async
async function safeF fetch() {
    try {
        let data = await fetchData("https://api.example.com")
        println(data)
    } catch (error) {
        println("Error:", error.message)
    }
}

# Parallel requests
async function fetchAll() {
    let urls = [
        "https://api1.example.com",
        "https://api2.example.com",
        "https://api3.example.com"
    ]
    
    let promises = urls.map(url => fetchData(url))
    let results = await Promise.all(promises)
    
    return results
}

# Run async code
fetchAll().then(data => {
    println("All data fetched:", data)
})

AI: LLM Integration

KLang has native support for Large Language Models. Connect to OpenAI, Anthropic, or local models:

import ai

# Simple LLM call
let response = ai.llm("What is the capital of France?")
println(response)

# With options
let result = ai.llm(
    "Explain quantum computing",
    {
        model: "gpt-4",
        temperature: 0.7,
        maxTokens: 500
    }
)

# Chat conversation
let chat = ai.chat()
chat.addMessage("user", "Hello!")
chat.addMessage("assistant", "Hi! How can I help?")
let reply = chat.send("Tell me about KLang")
println(reply)

# Using different providers
let openai = ai.llm("Question", {provider: "openai"})
let anthropic = ai.llm("Question", {provider: "anthropic"})
let ollama = ai.llm("Question", {provider: "ollama", model: "llama2"})

AI: Embeddings

import ai

# Generate embeddings
let text = "KLang is an AI-native programming language"
let embedding = ai.embed(text)
println("Embedding dimension:", len(embedding))

# Batch embeddings
let texts = [
    "First document",
    "Second document",
    "Third document"
]
let embeddings = ai.embedBatch(texts)

# Similarity search
let query = "Programming language"
let queryEmbed = ai.embed(query)

# Calculate cosine similarity
function cosineSimilarity(a, b) {
    let dot = 0
    let magA = 0
    let magB = 0
    
    for i in range(len(a)) {
        dot += a[i] * b[i]
        magA += a[i] ** 2
        magB += b[i] ** 2
    }
    
    return dot / (sqrt(magA) * sqrt(magB))
}

let similarity = cosineSimilarity(embedding, queryEmbed)
println("Similarity:", similarity)

AI: RAG Systems

import ai

# Create RAG system
let rag = ai.rag({
    documents: [
        "KLang is fast with 366M inst/sec",
        "KLang has native AI integration",
        "KLang uses a register-based VM"
    ],
    embedModel: "text-embedding-ada-002",
    llmModel: "gpt-4"
})

# Query with context
let answer = rag.query("How fast is KLang?")
println(answer)
# Output: KLang achieves 366 million instructions per second...

# Add more documents
rag.addDocument("KLang supports async/await")
rag.addDocument("KLang has 100+ opcodes")

# Query again
let answer2 = rag.query("What are KLang's features?")
println(answer2)

# Get sources
let result = rag.queryWithSources("Tell me about KLang performance")
println("Answer:", result.answer)
println("Sources:", result.sources)

AI: Autonomous Agents

import ai

# Create an agent
let agent = ai.agent({
    name: "CodeHelper",
    goal: "Help developers write better code",
    tools: ["search", "calculator", "code_executor"],
    llm: "gpt-4",
    memory: true
})

# Give the agent a task
agent.run("Write a function to calculate fibonacci numbers")

# Agent with custom tools
function customTool(input) {
    # Your custom logic
    return "Result: " + input
}

let agent2 = ai.agent({
    name: "CustomAgent",
    tools: [
        {
            name: "custom",
            function: customTool,
            description: "My custom tool"
        }
    ]
})

agent2.run("Use the custom tool with input: test")

# Multi-agent system
let researcher = ai.agent({name: "Researcher", goal: "Research topics"})
let writer = ai.agent({name: "Writer", goal: "Write articles"})

let research = researcher.run("Research AI programming languages")
let article = writer.run("Write an article based on: " + research)
println(article)

HTTP Client

import http

# GET request
let response = http.get("https://api.example.com/data")
println(response.status)      # 200
println(response.text)        # Response body
println(response.json())      # Parsed JSON

# POST request
let postData = {
    name: "John Doe",
    email: "john@example.com"
}

let response = http.post(
    "https://api.example.com/users",
    {
        headers: {
            "Content-Type": "application/json"
        },
        body: json.dumps(postData)
    }
)

# Custom request
let response = http.request(
    "PUT",
    "https://api.example.com/resource/123",
    {
        headers: {"Authorization": "Bearer token"},
        body: json.dumps({status: "active"})
    }
)

Performance

KLang is built for speed with a register-based virtual machine:

  • 366 million instructions per second
  • 2.7 nanoseconds per instruction
  • 10x faster than stack-based VMs
  • 256 registers per call frame
  • 100+ optimized opcodes
Tip: For maximum performance, use the VM's bytecode compilation or LLVM JIT compilation features.
